// 1. Configure OIDC Client (using 'openid-client' library)
const { Issuer } = require('openid-client');

const faydaIssuer = await Issuer.discover('https://id.gov.et/api');
const client = new faydaIssuer.Client({
  client_id: process.env.FAYDA_CLIENT_ID,
  token_endpoint_auth_method: 'private_key_jwt',
}, { 
  keys: [JSON.parse(process.env.MY_PRIVATE_JWK)] 
});

// 2. Callback Route
app.get('/auth/callback', async (req, res) => {
  const params = client.callbackParams(req);
  const tokenSet = await client.callback(
    process.env.REDIRECT_URI, 
    params, 
    { code_verifier: req.session.verifier }
  );

  // 3. Get Demographic Data (KYC)
  const userinfo = await client.userinfo(tokenSet.access_token);
  
  // 4. MERN: Upsert User in MongoDB
  const user = await User.findOneAndUpdate(
    { nationalId: userinfo.sub },
    { name: userinfo.name, lastLogin: new Date() },
    { upsert: true, new: true }
  );

  res.redirect(`http://localhost:3000/dashboard?token=${tokenSet.id_token}`);
});
// 1. TanStack Router Setup
const rootRoute = createRootRoute({ component: Layout });
const dashboardRoute = createRoute({
  getParentRoute: () => rootRoute,
  path: '/dashboard',
  beforeLoad: ({ location }) => {
    if (!isAuthenticated()) throw redirect({ to: '/login' });
  },
  component: Dashboard,
});

// 2. TanStack Query for Profile
function Dashboard() {
  const { data: profile, isLoading } = useQuery({
    queryKey: ['donorProfile'],
    queryFn: async () => {
      const res = await axios.get('/api/me');
      return res.data;
    }
  });

  if (isLoading) return <Loader />;

  return (
    <div>
      <h1>Welcome, {profile.name}</h1>
      <p>Verified National ID: {profile.nationalId}</p>
    </div>
  );
}
💡

Pro-Tip: donorLink should use the sub (Subject) field from Fayda as the unique identifier in MongoDB. Never store the actual National ID number as a plain string if you can help it; the sub is a persistent, unique hash that identifies the user without exposing sensitive PII unnecessarily.
